### 1. Refine `isComplete` logic for `runner` and `results` stages

**File:** `frontend/src/components/layout/WorkflowNavRail.tsx`
**Issue:** The current `isComplete` logic only checks for the existence of results or data availability, not necessarily successful completion. This can misrepresent the workflow stage status.

**Corrected code:**
```typescript
function isComplete(
  key: StageKey,
  selectedPopulationId: string,
  dataFusionResult: GenerationResult | null,
  portfolios: PortfolioListItem[],
  results: ResultListItem[],
): boolean {
  switch (key) {
    case "data-fusion":
      return isPopulationComplete(selectedPopulationId, dataFusionResult);
    case "portfolio":
      return portfolios.length > 0;
    case "runner":
      // Acknowledge meaningful work as at least one successful run
      return results.some((r) => r.status === "completed");
    case "results":
      // Acknowledge meaningful work as at least one successful run with data available
      return results.some((r) => r.status === "completed" && r.data_available);
  }
}
```

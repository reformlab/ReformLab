```python
def _validate_descriptor_fields(
    descriptor: dict[str, Any],
    pop_id: str,
) -> tuple[DataAssetOrigin, DataAssetAccessMode, DataAssetTrustStatus]:
    """Validate and extract descriptor fields, logging warnings for invalid values."""
    canonical_origin = descriptor.get("origin", "synthetic-public")
    access_mode = descriptor.get("access_mode", "bundled")
    trust_status = descriptor.get("trust_status", "exploratory")

    # Validate against literal types
    if canonical_origin not in DataAssetOrigin.__args__:
        logger.warning(
            "event=invalid_descriptor population=%s field=origin value=%s using_default",
            pop_id, canonical_origin,
        )
        canonical_origin = "synthetic-public"
    
    if access_mode not in DataAssetAccessMode.__args__:
        logger.warning(
            "event=invalid_descriptor population=%s field=access_mode value=%s using_default",
            pop_id, access_mode,
        )
        access_mode = "bundled"
    
    if trust_status not in DataAssetTrustStatus.__args__:
        logger.warning(
            "event=invalid_descriptor population=%s field=trust_status value=%s using_default",
            pop_id, trust_status,
        )
        trust_status = "exploratory"

    return (canonical_origin, access_mode, trust_status)
```

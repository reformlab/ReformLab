```typescript
/** Hash migration utilities — Story 26.1. */

import type { StageKey } from "@/types/workspace";
import { isValidStage } from "@/types/workspace";

/** Migrate legacy "engine" stage/hash to "scenario". Returns migrated value and whether migration occurred. */
export function migrateEngineStage(stage: string | null): { migrated: string; changed: boolean } {
  if (!stage) return { migrated: "policies", changed: false };
  const migrated = stage === "engine" ? "scenario" : stage;
  return { migrated, changed: migrated !== stage };
}

/** Migrate legacy #engine hash to #scenario. Returns migrated hash and whether migration occurred. */
export function migrateEngineHash(hash: string): { migrated: string; changed: boolean } {
  if (!hash.startsWith("#engine")) return { migrated: hash, changed: false };
  
  const parts = hash.slice(1).split("/");
  const migratedStage = parts[0] === "engine" ? "scenario" : parts[0];
  const subView = parts[1];
  const migrated = subView ? `#${migratedStage}/${subView}` : `#${migratedStage}`;
  
  return { migrated, changed: migratedStage !== parts[0] };
}

/** Migrate stage for localStorage persistence. Returns migrated stage if valid, null otherwise. */
export function migrateAndValidateStage(raw: string | null): StageKey | null {
  if (!raw) return null;
  
  const { migrated } = migrateEngineStage(raw);
  const valid = isValidStage(migrated);
  return valid ? migrated : null;
}
```

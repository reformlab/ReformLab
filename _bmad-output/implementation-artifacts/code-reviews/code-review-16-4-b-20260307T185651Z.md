# ... (inside setup cell)

def show(table: pa.Table, n: int = 12) -> str:
    """Return a readable table preview string.

    Args:
        table: The PyArrow Table to preview.
        n: The maximum number of rows to display.

    Returns:
        A formatted string representation of the table.
    """
    if table.num_rows == 0:
        return "  (empty table)"

    sliced = table.slice(0, n) if table.num_rows > n else table
    cols = sliced.column_names
    rows = [
        [str(sliced.column(c)[i].as_py()) for c in cols]
        for i in range(sliced.num_rows)
    ]
    widths = [max(len(c), *(len(r[j]) for r in rows)) for j, c in enumerate(cols)]
    header = "  ".join(c.ljust(w) for c, w in zip(cols, widths))
    sep = "  ".join("-" * w for w in widths)

    output_lines = []
    if cols:
        output_lines.append(header)
        output_lines.append(sep)
    for row in rows:
        output_lines.append("  ".join(v.ljust(w) for v, w in zip(row, widths)))
    if table.num_rows > n:
        output_lines.append(f"  ... ({table.num_rows - n} more rows)")
    
    return "\n".join(output_lines)

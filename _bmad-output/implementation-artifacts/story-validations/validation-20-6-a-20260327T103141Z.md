```typescript
function computePortfolioHash(portfolio: Portfolio): string {
  // Sort policies by name for order-independence
  const sorted = [...portfolio.policies].sort((a, b) => a.name.localeCompare(b.name));
  
  // Hash: name + JSON stringified parameters + resolution
  const input = sorted.map(p => 
    `${p.name}:${JSON.stringify(p.parameters)}:${portfolio.resolutionStrategy}`
  ).join('|');
  
  // Use SHA-256 (same as manifest hashing)
  return sha256(input);
}
```

**Suggested Addition:**

```typescript
// Cross-reference algorithm:
function buildMatrix(
  scenarios: WorkspaceScenario[],
  populations: Population[],
  runs: ResultMetadata[]
): ExecutionMatrix {
  const matrix: ExecutionMatrix = {};
  
  for (const scenario of scenarios) {
    matrix[scenario.id] = {};
    for (const pop of populations) {
      // Find run matching scenario + population
      const run = runs.find(r => 
        r.scenario_id === scenario.id && r.population_id === pop.id
      );
      
      matrix[scenario.id][pop.id] = run 
        ? { scenarioId: scenario.id, populationId: pop.id, status: run.status, runId: run.run_id }
        : { scenarioId: scenario.id, populationId: pop.id, status: ExecutionStatus.NOT_EXECUTED };
    }
  }
  
  return matrix;
```
